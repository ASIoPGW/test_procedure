\begin{equation}
\label{eq:singaussian}
    A \exp\left(-\frac{(t-t_0)^2}{\tau^2}\right) \cos ( 2 \pi f_0 (t-t_0))
\end{equation}
\begin{equation*}
   \tau = \frac{Q}{ \sqrt{2} \pi f_0}
\end{equation*}
